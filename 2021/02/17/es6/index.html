Let和const

// 用var声明的变量有两种作用域：全局作用域、函数作用域
// 用var声明的变量会有变量提升，会提升到当前作用域的最前面，函数也会提升，函数的声明的提升在变量提升之前

let b = 20;
b = 30;
// 用let和const声明的变量除了全局作用域和函数作用域之外，还有块级作用域
// 用let和const声明的变量在同一个作用域中只能有一个
// 用let和const声明的变量没有变量提升
// 用let声明的是变量，用const声明的是常量(const声明的变量赋值之后不可以修改内存地址)
// 在块级作用域中声明的变量，其变量的作用范围是当前的块，在块级外部无法访问

{

}
{

  function foo({ uname = “22”, uage = “33” } = {}) {
    console.log(uname, uage);
  }
  // foo({uname:”哈哈”,uage:111})

}

{
  //剩余参数 ：需要是函数参数的最后一个
  function foo(a, b, …reset) {
    console.log(a, b, reset);
  }
  foo(1, 2, 3, 3, 4, 5);
}

{
  //展开运算符展开数组  …arr1 表示将arr1数组中的每一个元素展开
  var arr1 = [1, 2, 3];
  var arr2 = [4, 5, 6];
  //   var arr3 = arr1.concat(arr2)
  var arr3 = […arr1, …arr2];
  console.log(arr3);

  //展开运算符展开对象
  var obj1 = {
    name: “xiaohei”,
  };
  var obj2 = {
    age: 222,
  };
  var obj3 = { …obj1, …obj2 };
  console.log(obj3);
}

{
  //箭头函数    (参数)=>{函数体}
  var foo = (params) => {
    console.log(params);
  };
  foo(“999”);

  //在使用箭头函数的时候，如果函数体只是一行代码，函数体的大括号可以省略，函数体的值就是函数的返回值
  var foo2 = () => 123;
  /**
   上面代码等效于：
   var foo2 = () => {
       return 123;
   }
   */
  console.log(foo2());

  //箭头函数中的this代表的含义是：声明箭头函数时所在作用域的this
  var obj = {
    name: “小明”,
    say: function () {
      //我们在say方法中声明了this，所以此时箭头函数中的this就是say方法中的this
      setTimeout(() => {
        console.log(this.name);
      }, 1000);
    },
    say2: function () {
      //定时器普通函数中的this是window
      setTimeout(function () {
        console.log(this.name);
      }, 1000);
    },
  };
  obj.say2();
  obj.say();

  
  //箭头函数中没有arguments参数列表,对应的我们可以使用剩余参数来替代arguments
  var test = (…reset)=>{
      console.log(reset)
  }
  test()

  //箭头函数没有原型，所以箭头函数不能new
  console.log(test.prototype,”xxxx”)

}

Symbol

{
  //Symbol的变量是唯一的，所以a1和a2永远不相等
  let a1 = Symbol();
  let a2 = Symbol();
  console.log(a1 === a2);
}

{
  //Symbol全局注册表是一个类似全局作用域的共享环境
  //Symbol.for()方法首先在全局Symbol注册表中搜索键为”a3”的Symbol是否存在，如果存在，直接返回已有的Symbol；否则，创建一个新的Symbol，并使用这个键在Symbol全局注册表中注册，随即返回新创建的Symbol。后续如果再传入同样的键调用Symbol.for()会返回相同的Symbol
  var a1 = Symbol.for(“aa”);
  var a2 = Symbol.for(“aa”);
  console.log(a1 === a2);
}

Promise

//解决异步问题的第一种方案：基于回调函数
//基于回调函数的方案可以解决异步问题，但是会造成很严重的回调地狱的问题

//在es6中有一个函数叫Promise，该函数可以帮我们解决回调地狱的问题

//Promise 本身表示一个对未来的承诺，反应的是未来某个时段的执行结果

//new Promise()需要入参一个函数，该函数是立马执行的，该函数有两个参数，分别是resolve和reject
//Promise本身不是异步的，只不过Promise中可以有异步任务
//Promise有三种状态，分别是Pending(进行中) Resolved(成功) 和 Rejected(失败)，当调用resolve()的时候，Promise状态会由Pending变为Resolved；当调用rejecte()的时候，Promise状态会由Pending变为Rejected
//一旦Promise状态发生改变之后，Promise的then方法就会执行了
//Promise的then方法可以入参两个函数，第一个函数表示Promise成功的回调，第二个函数表示Promise失败的回调//Promise对象会有一个then()方法
//Promise的then()方法的函数参数中会默认返回一个新的Promise对象，所以我们可以在后面继续.then，从而构成Promise链

//Promise捕获异常：当reject()方法调用之后，then方法的第二个函数没有写的时候，catch方法会捕获到reject的错误
//Promise捕获异常：当reject()方法调用之后，then方法的第二个函数写了的时候，catch方法不会执行
//Promise捕获异常：当reject()方法调用之后，then方法的第二个函数写了的时候,并且在函数里面throw错误之后，catch方法会执行
//所以一般Promise的then方法的第二个函数不用写，直接交给catch方法来处理异常

//Promise.all([p1,p2])方法是等到所有的promise对象有了结果之后才会执行后面的then
// Promise.all([p1, p2]).then(function (results) {
//     console.log(results);
// });

//Promise.race([p1,p2])方法是在任何一个promise对象有了结果之后就立马执行后面的then
Promise.race([p1, p2]).then(function (result) {
    console.log(result);
});

Async和await

//加async修饰的方法表示异步方法，该方法会默认返回一个Promise
//只有在asnyc修饰的方法中可以使用await
//async和await结合起来使用的时候可以将原来的异步任务变为现在的同步任务
async function read(){
    //await表示等待，后面需要跟一个Promise对象
    let r1 = await readData(“./data/a.txt”);
    console.log(r1)
    let r2 = await readData(“./data/b.txt”);
    console.log(r2)
    let r3 = await readData(“./data/c.txt”);
    console.log(r3)
}

//下面代码31行没有等到30行的代码执行完毕之后再执行，说明read()方法是异步的 （read方法外面是异步的，里面是同步的）
read();
read();

遍历的方法

//1.普通for循环

//2.数组的若干遍历方法  forEach  map  $.each

//3.for … in遍历 ： 遍历对象
var obj = {
  name: “张三”,
  age: 123,
};
for (var key in obj) {
  console.log(key, obj[key]);
}

//4. for … of遍历：迭代器的遍历方式
//凡是实现了迭代器接口的对象都可以使用for of来遍历(比如数组 Map Set都可以使用for of来遍历，但是Object对象不能使用for of遍历，因为Object对象没有实现迭代器接口)
var arr = [“11”, “试试”, “44”, “ksks”];
for (var item of arr) {
  console.log(item);
}

for (let [index, value] of [“a”, “b”, “c”].entries()) {
  console.log(index, value);
}

//遍历器（Iterator）是一种接口，为各种不同的数据结构提供统一的访问机制
{
  //获取元素的迭代对象
  let arr = [“hello”, “world”];
  //获取迭代器对象
  let map = arrSymbol.iterator;
  //迭代器对象next()方法用来获取对象的下一个元素
  console.log(map.next());
  console.log(map.next());
  console.log(map.next());
  //for of遍历方式内部就是调用迭代器对象的next方法
}

{
  //定义一个对象 例:
  let obj = {
    name: “张飞”,
    age: 23,
    sex: “男”,
  };
  //手动给object 部署 Symbol.iterator接口机制
  obj[Symbol.iterator] = function () {
    //求对象长度  对象方法
    var len = Object.keys(obj).length;
    var index = 0; //定义下标
    var key; //定义键
    var keyVal; //定义键值
    return {
      //返回next方法
      next: function () {
        key = Object.keys(obj)[index++]; //通过下边选中键
        keyVal = obj[key]; //对应的键值
        //当下边小于等于对象长度时  返回 迭代数据
        return index <= len
          ? { value: key + “:” + keyVal, done: false }
          : { value: undefined, done: true };
      },
    };
  };

  for (var key of obj) {
    console.log(key);
  }
}
