原型介绍

使用原型共享构造函数的所有实例公共的属性和方法

function Person(name, age) {
        this.name = name;
        this.age = age;
      }
      //在 Person 的原型上添加一个方法叫 say，该 say 方法可以被 Person 的所有实例对象共享
      Person.prototype.say = function () {
        console.log(this.name + “在说话”);
      };
      Person.prototype.country = “中国”;

var p1 = new Person(“张三”, 18);
      var p2 = new Person(“李四”, 22);

console.log(p1.say == p2.say);
      Person.prototype.country = “日本”;
      console.log(p2.country, p2.say);

// toString 方法是挂载到 Object 的 prototype 中的一个方法，该方法可以被 Object 的所有实例共享
      // console.log(p1.toString == p2.toString);

//什么样子的数据是需要写在原型中?
      //1.需要共享的数据就可以写原型中，比如需要共享的属性和方法
      //2.不需要共享的数据写在构造函数中

var str1 = new String(“abc”);
      var str2 = new String(“edf”);

//  给内置对象的原型添加公共的方法
      String.prototype.myreverse = function () {
        var newStr = “”;
        for (var i = this.length - 1; i >= 0; i–) {
          newStr += this[i];
        }
        return newStr;
      };

console.log(str1.myreverse());
      console.log(str2.myreverse());

构造函数 隐式原型 显示原型之间的关系

所有的函数都会有原型对象,prototype 就是函数的原型对象
      //———————————–
      //在函数的原型对象(prototype)中包含三个部分的信息:
      //1.  该函数所有实例公共的属性和方法
      //2.  构造函数，该构造函数就是当前函数本身
      //3.  隐式原型：proto，该隐式原型指向父函数的原型对象
      function Person(name, age) {
        this.name = name;
        this.age = age;
      }
      Person.prototype.country = “中国”;
      Person.prototype.say = function () {
        console.log(this.name + “在说话”);
      };
      Person.prototype.name = “哈哈”;
      console.log(Person.prototype);

var zhangsan = new Person(“张三”, 123);

//函数对象.constructor ===  当前函数本身
      console.log(zhangsan.constructor === Person);
      //函数.prototype.constructor ===  当前函数本身
      console.log(Person.prototype.constructor === Person);

//—————————————-
      //函数对象上包含两个部分的信息:
      //1.  实例属性    实例方法
      //2.  隐式原型(不是 w3c 标准) proto ，该隐式原型指向函数的原型对象
      console.log(zhangsan.proto === Person.prototype);
      // zhangsan.proto.say()  就相当于  Person.prototype.say()，此时如果 say 方法中有 this，该 this 代表当前的原型对象本身
      zhangsan.proto.say(); 
      // zhangsan.say()，此时如果 say()方法中有 this，该 this 代表 zhangsan 本身,会优先用 zhangsan 本身的成员变量/成员方法，如果 zhangsan 没有才会使用原型中的信息
      zhangsan.say()
      console.log(zhangsan)

自调用函数

// 自定义函数：声明函数的就调用函数
      (function () {
        var num = 10;
        console.log(“aa”);
      })();

      //自调用函数的作用：开启一个独立的作用域，以防止变量的命名污染
      (function () {
        function foo() {
          console.log(“bb”);
        }
        console.log(foo);
      })();
//我们在使用自调用函数的时候 一般前面要加; 以隔离
(function () {
   console.log(“1”);
})();
(function () {
   console.log(“2”);
})();
